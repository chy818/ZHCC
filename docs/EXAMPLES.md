### 循环示例

```xy
/**
 * 循环示例 - 打印九九乘法表
 */
函数 主(): 整数 {
    定义 可变 甲: 整数 = 1
    
    循环 甲 <= 9 {
        定义 可变 乙: 整数 = 1
        
        循环 乙 <= 甲 {
            打印整数(甲)
            打印文本("×")
            打印整数(乙)
            打印文本("=")
            打印整数(甲 * 乙)
            打印文本("  ")
            乙 = 乙 + 1
        }
        
        打印换行()
        甲 = 甲 + 1
    }
    
    返回 0
}
```

## 算法示例

### 斐波那契数列

```xy
/**
 * 斐波那契数列
 */
函数 斐波那契(数值: 整数): 整数 {
    若 数值 <= 1 则 {
        返回 数值
    }
    返回 斐波那契(数值 - 1) + 斐波那契(数值 - 2)
}

函数 主(): 整数 {
    定义 可变 索引: 整数 = 0
    
    打印文本("斐波那契数列前 10 项：")
    循环 索引 < 10 {
        打印整数(斐波那契(索引))
        打印文本(" ")
        索引 = 索引 + 1
    }
    
    返回 0
}
```

### 二分查找

```xy
/**
 * 二分查找算法
 */
函数 二分查找(数组: 整数[], 目标: 整数): 整数 {
    定义 左边界: 整数 = 0
    定义 右边界: 整数 = 数组长度(数组) - 1
    
    循环 左边界 <= 右边界 {
        定义 中间: 整数 = (左边界 + 右边界) / 2
        
        若 数组[中间] == 目标 则 {
            返回 中间
        } 否则 若 数组[中间] < 目标 则 {
            左边界 = 中间 + 1
        } 否则 {
            右边界 = 中间 - 1
        }
    }
    
    返回 -1
}
```

### 冒泡排序

```xy
/**
 * 冒泡排序算法
 */
函数 冒泡排序(数组: 整数[]): 整数[] {
    定义 可变 结果: 整数[] = 数组
    定义 长度: 整数 = 数组长度(结果)
    定义 可变 索引甲: 整数 = 0
    
    循环 索引甲 < 长度 - 1 {
        定义 可变 索引乙: 整数 = 0
        
        循环 索引乙 < 长度 - 1 - 索引甲 {
            若 结果[索引乙] > 结果[索引乙 + 1] 则 {
                // 交换元素
                定义 临时: 整数 = 结果[索引乙]
                结果[索引乙] = 结果[索引乙 + 1]
                结果[索引乙 + 1] = 临时
            }
            索引乙 = 索引乙 + 1
        }
        索引甲 = 索引甲 + 1
    }
    
    返回 结果
}
```

        索引 = 索引 + 1
    }
    
    返回 结果
}

### 二叉树

```xy
/**
 * 二叉树节点
 */
结构体 树节点 {
    数值: 整数,
    左子树: 树节点指针,
    右子树: 树节点指针
}

/**
 * 插入节点
 */
函数 插入节点(根节点: 树节点指针, 数值: 整数): 树节点指针 {
    若 根节点 == 空 则 {
        返回 创建树节点(数值)
    }
    
    若 数值 < 根节点.数值 则 {
        根节点.左子树 = 插入节点(根节点.左子树, 数值)
    } 否则 {
        根节点.右子树 = 插入节点(根节点.右子树, 数值)
    }
    
    返回 根节点
}
```

### 栈实现

```xy
/**
 * 栈结构体
 */
结构体 栈 {
    数据: 整数[],
    容量: 整数,
    栈顶: 整数
}

/**
 * 创建栈
 */
函数 创建栈(容量: 整数): 栈 {
    返回 栈 {
        数据: [],
        容量: 容量,
        栈顶: -1
    }
}

/**
 * 入栈
 */
函数 入栈(栈实例: 栈, 数值: 整数): 布尔 {
    若 栈实例.栈顶 >= 栈实例.容量 - 1 则 {
        返回 假  // 栈满
    }
    
    栈实例.栈顶 = 栈实例.栈顶 + 1
    栈实例.数据[栈实例.栈顶] = 数值
    返回 真
}

/**
 * 出栈
 */
函数 出栈(栈实例: 栈): 整数 {
    若 栈实例.栈顶 < 0 则 {
        返回 -1  // 栈空
    }
    
    定义 数值: 整数 = 栈实例.数据[栈实例.栈顶]
    栈实例.栈顶 = 栈实例.栈顶 - 1
    返回 数值
}
```

## 实用工具

### 数学工具

```xy
/**
 * 判断是否为质数
 */
函数 是质数(数值: 整数): 布尔 {
    若 数值 <= 1 则 {
        返回 假
    }
    
    定义 可变 索引: 整数 = 2
    
    循环 索引 * 索引 <= 数值 {
        若 数值 % 索引 == 0 则 {
            返回 假
        }
        索引 = 索引 + 1
    }
    
    返回 真
}

/**
 * 计算最大公约数
 */
函数 最大公约数(甲: 整数, 乙: 整数): 整数 {
    定义 可变 甲值 = 甲
    定义 可变 乙值 = 乙
    
    循环 乙值 != 0 {
        定义 临时: 整数 = 乙值
        乙值 = 甲值 % 乙值
        甲值 = 临时
    }
    
    返回 甲值
}
```

 * 数组去重

函数 数组去重(数组: 整数[]): 整数[] {

## 完整示例

### 简单计算器

```xy
/**
 * 简单计算器
 */
函数 计算(甲: 整数, 运算符: 文本, 乙: 整数): 整数 {
    若 运算符 == "+" 则 {
        返回 甲 + 乙
    } 否则 若 运算符 == "-" 则 {
        返回 甲 - 乙
    } 否则 若 运算符 == "*" 则 {
        返回 甲 * 乙
    } 否则 若 运算符 == "/" 则 {
        若 乙 == 0 则 {
            打印文本("错误：除数不能为零")
            返回 0
        }
        返回 甲 / 乙
    } 否则 {
        打印文本("错误：未知运算符")
        返回 0
    }
}

函数 主(): 整数 {
    打印文本("简单计算器示例")
    打印整数(计算(10, "+", 5))
    打印整数(计算(10, "-", 3))
    打印整数(计算(10, "*", 2))
    打印整数(计算(10, "/", 2))
    返回 0
}
```
